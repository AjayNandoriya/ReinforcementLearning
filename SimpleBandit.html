steps = 1000

eps = [0.1]
alpha = 0.1

rewards,optAct = bandit.runSimpleBandit(R,steps,runs,eps,0,0)
r,o = bandit.runSimpleBandit(R,steps,runs,eps,alpha,0)
rewards = [rewards[0],r[0]]
optAct  = [optAct[0],o[0]]

fig,ax = plt.subplots(1,2,figsize=(15,5))
lgd = []
for i,e in enumerate(rewards): 
    ax[0].plot(range(steps),rewards[i]);
    ax[1].plot(range(steps),optAct[i]); 
ax[0].legend(["Sampled Average","Constant Step Size"])
ax[0].grid(True);
ax[0].set_ylabel('Average Reward');
